% GFR versus GDC
/*apare S->e dar S nu mai exista in partea dreapta a nici unei alte productii, deci este GDC*/
grammar1 ::= Grammar (
	nonterminals :: { S, B, C } ;
	alphabet :: { a, b, c } ;
	start_symbol :: S ;
	
	S -> a&a&B&C | a&b&C | e ;
	C&B -> B&C ;
	b&B -> b&b ;
	b&C -> b&c ;
	c&B -> B&c ;
	c&C -> c&c ;
) ;;

/*Exemplu de gramatica fara restrictii*/
/*apare S->e si S apare si in partea dreapta a unei alte productii=> gramatica fara restrictii*/
grammar1_v1 ::= Grammar (
	nonterminals :: { S, B, C } ;
	alphabet :: { a, b, c } ;
	start_symbol :: S ;
	
	S -> a&a&B&C | a&b&C | e ;
	C&B -> B&C ;
	b&B -> b&b ;
	b&C -> b&c ;
	c&B -> B&c&S ;
	c&C -> c&c ;
) ;;

% Automatul pentru w w^r unde w este din {a, b}*
automat_cu_stari_finale ::= PushDownAutomata (
	alphabet :: { a, b } ;
	stack_alphabet :: { Z0, a, b } ;
	states :: { q0, q1, q2, q3 } ;
	variable y ::= { a, b } ;
	
	q0 -> ( 
		a, Z0 / a&Z0 -> q1 ;
		b, Z0 / b&Z0 -> q1 ;
	) ;
	
	q1 -> ( 
		a, a / a&a -> q1 ;
		b, b / b&b -> q1 ;
		a, b / a&b -> q1 ;
		b, a / b&a -> q1 ;
		a, a / e -> q2 ; 
		b, b / e -> q2 ;
	) ;
	
	% Aceasta linie reprezinta un comentariu
	
	q2 -> (
		b, b / e -> q3 ;
		a, a / e -> q2 ;
		y, Z0 / Z0 -> q3 ;
	) ;
	
	final_states :: { q0, q3 } ;
	initial_state :: q0 ;
) ;;


% GIC
/*w din {a, b}* | |w| = 2k*/
grammar2 ::= Grammar (
	nonterminals :: { S } ;
	alphabet :: { a, b } ;
	start_symbol :: S ;
	
	%gramatica independenta de context
	S -> a&S&b | a&S&a | b&S&a | b&S&b | e ;
) ;;

/*GFR*/
grammar3 ::= Grammar (
	nonterminals :: { S, B, C } ;
	alphabet :: { a, b, c } ;
	start_symbol :: S ;
	
	S -> a&a&B&C | e | a&b&C ;
	C&B -> B&C ;
	b&B -> b&b ;
	b&C -> b&c ;
	/*S apare in partea dreapta a unei productii->GFR*/
	c&B -> S&B&c ;
	c&C -> c&c ;
) ;;

% Definitie automat cu stiva
automat_cu_stiva ::= PushDownAutomata (
	alphabet :: { a, b } ; %comentariu
	states :: { q0, q1, q2, q3, q4 } ;
	initial_state :: q0 ; 
	stack_alphabet :: { Z0, a, b } ;
	final_states :: {q1, q3, q4 } ; 
	
	q0 -> ( 
		a, Z0 / a&Z0 -> q1 ; %comentariu
		b, Z0 / Z0 -> q4 ;
	) ;
	
	q1 -> ( 
		a, a / a&a -> q1 ; /*comentariu ignorat*/
		b, a / e -> q2 ;
	) ;
	
	q2 -> (
		b, a / e -> q2 ;
		b, Z0 / Z0 -> q4 ;
		e, a / a -> q3 ;
	) ;
	
	q4 -> (
		b, Z0 / Z0 -> q4 ;
	) ;
	
	/*acesta este un comentariu*/
) ;;
