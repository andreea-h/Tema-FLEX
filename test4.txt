	nonterminals :: { S, B, C } ;
	alphabet :: { a, b, c } ;
	start_symbol :: S ;
	
	S -> a&a&B&C | a&b&C | e ;
	C&B -> B&C ;

	c&C -> c&c ;
) ;;
